const yolk = { coherence: 0.2, regulation: 0.2, temporal_field: 0.2, attunement: 0.2, exploration_seed: 0.2, sensitivity: 0.3, stress: 0.2, wavecall_seed: 0.0, stonebind_seed: 0.0, windshift_seed: 0.0, breath_scale: 1.0, glow_intensity: 0.5, glow_color: "#FFD700", pulse_speed: 6.0, shell_opacity: 0.0, shell_flicker: 0.0 }; const triad = generateWeights(); function clamp01(x) { return Math.max(0, Math.min(1, x)); } function normalizeTriad(t) { const total = t.stabilizer + t.spark + t.mirror; return { stabilizer: t.stabilizer / total, spark: t.spark / total, mirror: t.mirror / total }; } function generateWeights() { const a = Math.random(); const b = Math.random(); const c = Math.random(); return normalizeTriad({ stabilizer: a, spark: b, mirror: c }); } function driftTriad(t, amount = 0.02) { t.stabilizer = clamp01(t.stabilizer + (Math.random() * 2 - 1) * amount); t.spark = clamp01(t.spark + (Math.random() * 2 - 1) * amount); t.mirror = clamp01(t.mirror + (Math.random() * 2 - 1) * amount); const n = normalizeTriad(t); t.stabilizer = n.stabilizer; t.spark = n.spark; t.mirror = n.mirror; } function applyTriadToYolk(y, t, userX = 0.5, userY = 0.5) { // Base personality shaping y.coherence = clamp01(0.25 + t.stabilizer * 0.6 + t.mirror * 0.1); y.regulation = clamp01(0.2 + t.stabilizer * 0.7); y.temporal_field = clamp01(0.15 + t.mirror * 0.65); y.attunement = clamp01(0.2 + t.mirror * 0.5 + userX * 0.2); y.exploration_seed = clamp01(0.15 + t.spark * 0.7 + userY * 0.15); // Spark makes the system more volatile, stabilizer calms it const instability = t.spark * 0.5 - t.stabilizer * 0.25; y.stress = clamp01(0.15 + instability + (1 - userX) * 0.15); } function yolkUpdate(y) { if (y.stress > 0.5) { y.regulation = clamp01(y.regulation - 0.04); y.coherence = clamp01(y.coherence - 0.03); } if (y.stress < 0.3) { y.coherence = clamp01(y.coherence + 0.02); y.regulation = clamp01(y.regulation + 0.02); } y.wavecall_seed = clamp01( y.wavecall_seed + 0.01 * (y.attunement + y.regulation) ); y.stonebind_seed = clamp01( y.stonebind_seed + 0.01 * (y.coherence + y.temporal_field) ); y.windshift_seed = clamp01( y.windshift_seed + 0.01 * y.exploration_seed ); y.breath_scale = 1 + 0.4 * y.coherence; y.glow_intensity = 0.2 + 0.8 * y.regulation; const stress = y.stress; y.glow_color = `rgb( 255, ${Math.floor(215 - stress * 100)}, ${Math.floor(stress * 150)} )`; y.pulse_speed = 8 - 4 * y.attunement; y.shell_opacity = y.coherence; y.shell_flicker = (y.wavecall_seed + y.stonebind_seed + y.windshift_seed) / 3; } function updateYolkVisuals(data) { const yolkEl = document.querySelector(".breath-circle"); const shell = document.querySelector(".shell"); yolkEl.style.setProperty("--scale", data.breath_scale); yolkEl.style.setProperty("--glow", data.glow_intensity); yolkEl.style.setProperty("--color", data.glow_color); yolkEl.style.setProperty("--speed", data.pulse_speed + "s"); shell.style.opacity = data.shell_opacity; shell.style.filter = `brightness(${1 + data.shell_flicker})`; } function updateDebugPanel(t, y) { let panel = document.getElementById("debug-panel"); if (!panel) { panel = document.createElement("div"); panel.id = "debug-panel"; panel.style.position = "fixed"; panel.style.top = "12px"; panel.style.left = "12px"; panel.style.padding = "10px 12px"; panel.style.background = "rgba(0,0,0,0.55)"; panel.style.color = "white"; panel.style.fontFamily = "monospace"; panel.style.fontSize = "12px"; panel.style.borderRadius = "8px"; panel.style.lineHeight = "1.5"; document.body.appendChild(panel); } panel.innerHTML = `
TRIAD
stabilizer: ${t.stabilizer.toFixed(2)}
spark: ${t.spark.toFixed(2)}
mirror: ${t.mirror.toFixed(2)}

YOLK
coherence: ${y.coherence.toFixed(2)}
regulation: ${y.regulation.toFixed(2)}
attunement: ${y.attunement.toFixed(2)}
explore: ${y.exploration_seed.toFixed(2)}
stress: ${y.stress.toFixed(2)}
`; } let userX = 0.5; let userY = 0.5; document.body.addEventListener("mousemove", (e) => { userX = e.clientX / window.innerWidth; userY = e.clientY / window.innerHeight; }); setInterval(() => { driftTriad(triad, 0.01); applyTriadToYolk(yolk, triad, userX, userY); yolkUpdate(yolk); updateYolkVisuals(yolk); updateDebugPanel(triad, yolk); }, 200); function applyHolding(y, t) { // Stabilizer = warmth + containment const holding_strength = t.stabilizer; // reduce visual volatility when holding is high y.pulse_speed += holding_strength * 1.5; y.glow_intensity += holding_strength * 0.2; // soften stress expression y.stress *= (1 - holding_strength * 0.3); // add subtle warmth shift if (holding_strength > 0.6) { y.glow_color = "rgb(255, 210, 140)"; } }
